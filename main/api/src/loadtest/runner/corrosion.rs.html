73impl CorrosionHandle {
74    async fn setup(params: CorrosionParams) -> eyre::Result<Self> {
75        let temp = tempfile::TempDir::new()?;
76        let root = camino::Utf8Path::from_path(temp.path()).expect("non-utf8 temp path");
77        let db_path = root.join("db.sqlite");
78        let sub_path = root.join("subs");
79        // Pre-create so corrosion's async rmdir wins the race against TempDir's drop.
80        std::fs::create_dir_all(sub_path.as_std_path())?;
81
82        let maintenance = params.max_pages.map(|max_page_count| DBMaintenance {
83            limits: DBLimits {
84                max_page_count: Some(max_page_count),
85                ..Default::default()
86            },
87            ..DBMaintenance::default()
88        });
89
90        let db = InitializedDb::setup(&db_path, SCHEMA, maintenance).await?;
91        let subs = SubsManager::default();
92        let btx = BroadcastingTransactor::new(
93            db.actor_id,
94            db.clock.clone(),
95            db.pool.clone(),
96            subs.clone(),
97            None,
98        )
99        .await?;
100
101        let trip = Trip::new();
102        let pubsub_ctx = PubsubContext {
103            schema: db.schema,
104            subs,
105            pool: db.pool,
106            cache: Default::default(),
107            path: sub_path,
108            tripwire: trip.tripwire(),
109        };
110
111        let metrics = Metrics::new(quilkin::metrics::registry());
112        let server = Server::new_unencrypted(
113            "[::1]:0".parse::<SocketAddr>()?,
114            btx,
115            pubsub_ctx,
116            metrics.clone(),
117        )?;
118        let addr = server.local_addr();
119
120        Ok(Self {
121            addr,
122            server,
123            trip,
124            metrics,
125            _temp: temp,
126        })
127    }
128
129    fn addr(&self) -> SocketAddr {
130        self.addr
131    }
132
133    async fn shutdown(self) {
134        self.server.shutdown("loadtest complete").await;
135        self.trip.shutdown().await;
136    }
137}

139pub async fn run(params: CorrosionParams) -> eyre::Result<CorrosionResults> {
140    let handle = CorrosionHandle::setup(params).await?;
141    let addr = handle.addr();
142    let metrics = handle.metrics.clone();
143
144    let start = Instant::now();
145
146    let handles: Vec<_> = (0..params.writers)
147        .map(|i| {
148            let m = metrics.clone();
149            tokio::spawn(async move { run_writer(i, addr, m, params).await })
150        })
151        .collect();
152
153    let mut agg = WriterStats::default();
154    for h in handles {
155        match h.await {
156            Ok(Ok(ws)) => {
157                agg.attempted += ws.attempted;
158                agg.succeeded += ws.succeeded;
159                agg.errors_transport += ws.errors_transport;
160                agg.errors_exec += ws.errors_exec;
161                agg.latencies_ms.extend(ws.latencies_ms);
162            }
163            Ok(Err(e)) => tracing::error!(%e, "writer task failed"),
164            Err(e) => tracing::error!(%e, "writer task panicked"),
165        }
166    }
167
168    let elapsed = start.elapsed();
169    handle.shutdown().await;
170
171    agg.latencies_ms.sort_unstable_by(f64::total_cmp);
172    let stalls = agg.latencies_ms.iter().filter(|&&ms| ms >= 1000.0).count() as u64;
173    let (p50_ms, p95_ms, p99_ms, max_ms) = compute_percentiles(&agg.latencies_ms);
174
175    Ok(CorrosionResults {
176        attempted: agg.attempted,
177        succeeded: agg.succeeded,
178        errors_transport: agg.errors_transport,
179        errors_exec: agg.errors_exec,
180        actual_tps: agg.attempted as f64 / elapsed.as_secs_f64(),
181        duration: elapsed,
182        p50_ms,
183        p95_ms,
184        p99_ms,
185        max_ms,
186        stalls,
187    })
188}
189
190fn compute_percentiles(sorted: &[f64]) -> (f64, f64, f64, f64) {
191    if sorted.is_empty() {
192        return (0.0, 0.0, 0.0, 0.0);
193    }
194    let pct = |p: f64| {
195        let idx = ((p / 100.0) * sorted.len() as f64) as usize;
196        sorted[idx.min(sorted.len() - 1)]
197    };
198    (pct(50.0), pct(95.0), pct(99.0), *sorted.last().unwrap())
199}

201async fn run_writer(
202    id: u32,
203    addr: SocketAddr,
204    metrics: Metrics,
205    params: CorrosionParams,
206) -> eyre::Result<WriterStats> {
207    let client = Client::connect_insecure(addr, metrics).await?;
208    let icao: IcaoCode = "LOAD".parse().expect("valid ICAO");
209    let mc = MutationClient::connect(client, 0, icao).await?;
210
211    let interval = (params.rate > 0).then(|| Duration::from_nanos(1_000_000_000 / params.rate));
212    let mut stats = WriterStats {
213        latencies_ms: Vec::with_capacity(params.transactions as usize),
214        ..Default::default()
215    };
216
217    for seq in 0u64..params.transactions {
218        if let Some(iv) = interval {
219            // Schedule from now so a slow transaction doesn't trigger catch-up bursts.
220            tokio::time::sleep(iv).await;
221        }
222
223        let upserts: Vec<ServerUpsert> = (0..params.batch)
224            .map(|b| {
225                let flat = seq * params.batch as u64 + b as u64;
226                let endpoint = Endpoint::new(
227                    AddressKind::Ip(IpAddr::V4(Ipv4Addr::new(
228                        10,
229                        (id % 256) as u8,
230                        ((flat >> 8) & 0xff) as u8,
231                        (flat & 0xff) as u8,
232                    ))),
233                    7777,
234                );
235                ServerUpsert {
236                    endpoint,
237                    icao,
238                    tokens: TokenSet::default(),
239                }
240            })
241            .collect();
242
243        stats.attempted += 1;
244        let t0 = Instant::now();
245        match mc.transactions(&[ServerChange::Upsert(upserts)]).await {
246            Ok(resp) => {
247                let ms = t0.elapsed().as_secs_f64() * 1000.0;
248                if resp
249                    .results
250                    .iter()
251                    .any(|r| matches!(r, ExecResult::Error { .. }))
252                {
253                    stats.errors_exec += 1;
254                } else {
255                    stats.succeeded += 1;
256                    stats.latencies_ms.push(ms);
257                }
258            }
259            Err(e) => {
260                tracing::debug!(%e, writer = id, seq, "transaction failed");
261                stats.errors_transport += 1;
262            }
263        }
264    }
265
266    mc.shutdown().await;
267    Ok(stats)
268}

270#[cfg(test)]
271mod tests {
272    use super::*;
273
274    fn params(writers: u32, transactions: u64) -> CorrosionParams {
275        CorrosionParams {
276            writers,
277            transactions,
278            rate: 0,
279            batch: 1,
280            max_pages: None,
281        }
282    }
283
284    #[tokio::test(flavor = "multi_thread")]
285    async fn smoke() {
286        let r = run(params(1, 10)).await.expect("corrosion loadtest failed");
287        assert_eq!(
288            r.succeeded, 10,
289            "all transactions succeeded (transport={}, exec={})",
290            r.errors_transport, r.errors_exec
291        );
292    }
293
294    /// Single-writer p50 must stay under 20 ms.
295    #[tokio::test(flavor = "multi_thread")]
296    async fn latency_single_writer() {
297        let r = run(params(1, 50)).await.expect("corrosion loadtest failed");
298        assert_eq!(
299            r.succeeded, 50,
300            "all transactions succeeded (transport={}, exec={})",
301            r.errors_transport, r.errors_exec
302        );
303        eprintln!(
304            "[single writer]  p50={:.2}ms p95={:.2}ms p99={:.2}ms stalls={}",
305            r.p50_ms, r.p95_ms, r.p99_ms, r.stalls
306        );
307        assert!(r.p50_ms < 20.0, "p50 {:.1}ms exceeds 20ms", r.p50_ms);
308    }
309
310    /// 4-writer p50 must stay within 4× the single-writer baseline.
311    #[tokio::test(flavor = "multi_thread")]
312    async fn latency_concurrent_writers() {
313        let single = run(params(1, 30)).await.expect("single writer failed");
314        let multi = run(params(4, 30)).await.expect("multi writer failed");
315
316        assert_eq!(
317            single.succeeded, 30,
318            "single: transport={} exec={}",
319            single.errors_transport, single.errors_exec
320        );
321        assert_eq!(
322            multi.succeeded, 120,
323            "multi: transport={} exec={}",
324            multi.errors_transport, multi.errors_exec
325        );
326
327        eprintln!(
328            "[1 writer]  p50={:.2}ms  [4 writers]  p50={:.2}ms  ratio={:.1}x",
329            single.p50_ms,
330            multi.p50_ms,
331            multi.p50_ms / single.p50_ms.max(0.01)
332        );
333
334        assert!(
335            multi.p50_ms < single.p50_ms * 4.0 + 5.0,
336            "4-writer p50 {:.1}ms is more than 4× single-writer p50 {:.1}ms — write serialisation has grown",
337            multi.p50_ms,
338            single.p50_ms
339        );
340    }
